// ---------------------------------------------------
// General configuration
// ---------------------------------------------------

config {
	// Tries to clean up the output to make it XHTML compliant
	#xhtml_cleaning = all
	// Generates a document type declaration
	doctype = html5
	// Set MSIE to standard compliance mode
	doctypeSwitch = 1

	// Include Boilerplate handling for IE browsers
	htmlTag_stdWrap {
		setContentToCurrent = 1
		cObject = COA
		cObject {
			10 = LOAD_REGISTER
			10 {
				newLine.char = 10
				tagEnd {
					current = 1
					split.max = 2
					split.token = <html
					split.returnKey = 1
				}
			}
			15 = TEXT
			15.value = <!-- paulirish.com/2008/conditional-stylesheets-vs-css-hacks-answer-neither/ -->
			20 = TEXT
			20.value = <!--[if lt IE 7]> <html class="no-js ie6 oldie"{register:tagEnd} <![endif]-->
			20.wrap = |{register:newLine}
			20.insertData = 1
			30 < .20
			30.value = <!--[if IE 7]> <html class="no-js ie7 oldie"{register:tagEnd} <![endif]-->
			40 < .20
			40.value = <!--[if gte IE 9]> <html class="no-js ie9 ie10 oldie"{register:tagEnd} <![endif]-->
			50 < .20
			50.value = <!--[if (gte IE 9)|!(IE)]> <!--><html class="no-js"{register:tagEnd} <!--<![endif]-->
			90 = RESTORE_REGISTER
		}
	}

	// Prefix local anchors, necessary to make realurl work properly
	prefixLocalAnchors = 0
	// Add slash as prefix to relative path
	absRefPrefix = /
	// If set it will try to render a meaningful prefix before temporary image files
	meaningfulTempFilePrefix = 1
	// Suppress auto-rendering of page-title
	noPageTitle = 2
	// Allow inclusion of content from other page-trees with different domains
	content_from_pid_allowOutsideDomain = 1
	// check if link points to a page within the current rootline. If not and domain record is found, then TYPO3 will add other domain to link
	typolinkCheckRootline = 1
	// Suppress default JavaScript
	removeDefaultJS = external
	// Move Inline-Styles to temporary files, renders a much cleaner header
	inlineStyle2TempFile = 1
	// Remove CSS generated by _CSS_DEFAULT_STYLE configuration of extensions 
	removeDefaultCss = 1
	// bundles JS and CSS files
	concatenateJsAndCss = 1
	// all JavaScriptwill be moved to the bottom of the HTML document
	moveJsFromHeaderToFooter = 1
	// inline or externalized JavaScript will be minified
	minifyJS = 1
	// Surpress comments in document source-code
	disablePrefixComment = 1
	// Enable realurl generation
	tx_realurl_enable = {$portal.configuration.general.enableRealURL}
	// HTTP_GET_VARS, which should be passed on with links in TYPO3
	linkVars = L(0-6),style(0-1)
	// Avoids generation of links that contain the same parameter twice or even more often
	uniqueLinkVars = 1
	// Avoids internal website-stats as they slow down the site
	stat = 0
	admPanel = {$portal.configuration.admPanel}
	// Disables the border attributes in image tags
	disableImgBorderAttr = 1
	// enable browser cache-headers
	sendCacheHeaders = 1
	// Set path to HTML-Template for rendering of the PAGE-object
	pageRendererTemplateFile = typo3conf/ext/masterconfig/res/core/tslib_page_frontend.html
}

// Disable FE-cache if a BE-User is logged in
[globalVar = TSFE : beUserLogin> 0]
